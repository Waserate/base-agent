> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Integrate AAVE v3 Base chain — 5 supply platforms + 8 borrow platforms — into base-agent using the same open/close/health pattern as existing Compound/Moonwell/Fluid modules.

**Architecture:** Two new modules: `aave_supply.py` (Pool.supply + Pool.withdraw, type=`aave_supply`) and `aave_borrow.py` (supply collateral → borrow → release ETH, type=`aave_borrow`). Both call AAVE Pool directly. Health check via `Pool.getUserAccountData()`. All token acquisition/release handled by existing `swap.py`.

**Tech Stack:** Python, web3.py, AAVE v3 Pool (`0xA238Dd80C259a72e81d7e4664a9801593F98d1c5`), existing executor.py/swap.py/state.py/agent.py patterns.

```python
"""
aave_supply.py — AAVE v3 supply module for Base chain.

Supports 5 platforms (ERC20 supply → aToken):
  aave_usdc   — supply USDC   (3.20% APY)
  aave_weth   — supply WETH   (1.56% APY)
  aave_cbbtc  — supply cbBTC  (0.03% APY)
  aave_wsteth — supply wstETH (0.00% APY — farming only)
  aave_eurc   — supply EURC   (1.69% APY)

Platform config keys (contracts.json):
  token_address  : ERC20 underlying asset address
  atoken_address : AAVE aToken address (received after supply)
  type           : "aave_supply"

State: stores supplied amount_wei. Withdraw calls Pool.withdraw(MAX_UINT256) to redeem all aTokens.

AAVE v3 Pool Base: 0xA238Dd80C259a72e81d7e4664a9801593F98d1c5
"""

import os, time, logging
from web3 import Web3
from dotenv import load_dotenv
import executor

load_dotenv()
log = logging.getLogger(__name__)

DRY_RUN   = os.getenv('DRY_RUN', '').lower() in ('1', 'true', 'yes')
POOL_ADDR = '0xA238Dd80C259a72e81d7e4664a9801593F98d1c5'
MAX_UINT256 = 2**256 - 1

_POOL_ABI = [
    {'name': 'supply', 'type': 'function', 'stateMutability': 'nonpayable',
     'inputs': [
         {'name': 'asset',         'type': 'address'},
         {'name': 'amount',        'type': 'uint256'},
         {'name': 'onBehalfOf',    'type': 'address'},
         {'name': 'referralCode',  'type': 'uint16'},
     ], 'outputs': []},
    {'name': 'withdraw', 'type': 'function', 'stateMutability': 'nonpayable',
     'inputs': [
         {'name': 'asset',  'type': 'address'},
         {'name': 'amount', 'type': 'uint256'},
         {'name': 'to',     'type': 'address'},
     ], 'outputs': [{'name': '', 'type': 'uint256'}]},
]

_ATOKEN_ABI = [
    {'name': 'balanceOf', 'type': 'function', 'stateMutability': 'view',
     'inputs': [{'name': 'account', 'type': 'address'}],
     'outputs': [{'name': '', 'type': 'uint256'}]},
]


def _pool():
    return executor.w3.eth.contract(
        address=Web3.to_checksum_address(POOL_ADDR), abi=_POOL_ABI
    )


def supply(asset_addr: str, amount_wei: int) -> str:
    """Approve Pool + supply asset. Returns tx hash."""
    executor._guard()
    asset_addr = Web3.to_checksum_address(asset_addr)
    pool_addr  = Web3.to_checksum_address(POOL_ADDR)
    executor._approve_if_needed(asset_addr, pool_addr, amount_wei)
    pool = _pool()
    tx = pool.functions.supply(
        asset_addr, amount_wei, executor.WALLET, 0
    ).build_transaction(executor._tx_params())
    tx['gas'] = executor._gas_limit(tx)
    return executor._send(tx)


def withdraw_all(asset_addr: str) -> str:
    """Withdraw all aTokens for asset. Returns tx hash."""
    executor._guard()
    asset_addr = Web3.to_checksum_address(asset_addr)
    pool = _pool()
    tx = pool.functions.withdraw(
        asset_addr, MAX_UINT256, executor.WALLET
    ).build_transaction(executor._tx_params())
    tx['gas'] = executor._gas_limit(tx)
    return executor._send(tx)


def get_atoken_balance(atoken_addr: str) -> int:
    """Returns current aToken balance (underlying units, includes accrued interest)."""
    c = executor.w3.eth.contract(
        address=Web3.to_checksum_address(atoken_addr), abi=_ATOKEN_ABI
    )
    return c.functions.balanceOf(executor.WALLET).call()
```

```python
     'inputs': [
         {'name': 'asset',           'type': 'address'},
         {'name': 'useAsCollateral', 'type': 'bool'},
     ], 'outputs': []},
    {'name': 'getUserAccountData', 'type': 'function', 'stateMutability': 'view',
     'inputs': [{'name': 'user', 'type': 'address'}],
     'outputs': [
         {'name': 'totalCollateralBase',        'type': 'uint256'},
         {'name': 'totalDebtBase',              'type': 'uint256'},
         {'name': 'availableBorrowsBase',       'type': 'uint256'},
         {'name': 'currentLiquidationThreshold','type': 'uint256'},
         {'name': 'ltv',                        'type': 'uint256'},
         {'name': 'healthFactor',               'type': 'uint256'},  # 1e18 scale
     ]},
]

_VDEBT_ABI = [
    {'name': 'balanceOf', 'type': 'function', 'stateMutability': 'view',
     'inputs': [{'name': 'account', 'type': 'address'}],
     'outputs': [{'name': '', 'type': 'uint256'}]},
]


def _pool():
    return executor.w3.eth.contract(
        address=Web3.to_checksum_address(POOL_ADDR), abi=_POOL_ABI
    )


# ── State encoding ─────────────────────────────────────────────────────────────

def encode_state(coll_token: str, coll_wei: int, borrow_token: str, borrow_wei: int) -> str:
    return f"{coll_token}:{coll_wei}||{borrow_token}:{borrow_wei}"


def decode_state(encoded: str) -> tuple:
    """Returns (coll_token, coll_wei, borrow_token, borrow_wei)."""
    coll_part, borrow_part = encoded.split('||')
    coll_token,  coll_wei_s  = coll_part.split(':')
    borrow_token, borrow_wei_s = borrow_part.split(':')
    return coll_token, int(coll_wei_s), borrow_token, int(borrow_wei_s)


# ── Health check ────────────────────────────────────────────────────────────────

def check_health(encoded_state: str, p: dict) -> float:
    """Returns AAVE account health factor (aggregate, not per-position)."""
    data = _pool().functions.getUserAccountData(executor.WALLET).call()
    hf = data[5] / 1e18
    return hf


# ── Borrow amount calculation ──────────────────────────────────────────────────

def _compute_borrow_wei(p: dict, ltv: float) -> int:
    """
    Compute borrow amount in borrow_token units:
      borrow_usd = collateral_usd × ltv
      borrow_wei = borrow_usd / borrow_price × 10^borrow_decimals
    """
    coll_token    = p['collateral_token']
    coll_wei      = p['collateral_amount_wei']
    coll_decimals = p['collateral_decimals']
    borrow_token  = p['borrow_token']
    borrow_dec    = p['borrow_decimals']

    coll_price_usd   = executor.get_token_usd_price(coll_token)
    borrow_price_usd = executor.get_token_usd_price(borrow_token)

    coll_usd   = (coll_wei / 10**coll_decimals) * coll_price_usd
    borrow_usd = coll_usd * ltv
    borrow_amt = borrow_usd / borrow_price_usd * 10**borrow_dec
    return int(borrow_amt)


# ── Open borrow ────────────────────────────────────────────────────────────────

def open_borrow(p: dict) -> tuple:
    """
    Open an AAVE borrow position.
    Returns (encoded_state, tx_hash_of_borrow).

    Steps:
      1. Acquire collateral (wrap WETH or swap ETH→token)
      2. Approve Pool for collateral
      3. Pool.supply(collateral, amount, wallet, 0)
      4. Pool.setUserUseReserveAsCollateral(collateral, True)
      5. Compute borrow_wei from collateral_usd × random_ltv
      6. Pool.borrow(borrow_asset, borrow_wei, 2, 0, wallet)
      7. Swap borrow_token → ETH (or unwrap WETH)
      8. Return encode_state + final borrow txh
    """
    executor._guard()
    pool = _pool()
    pool_addr = Web3.to_checksum_address(POOL_ADDR)

    coll_token  = p['collateral_token']
    coll_addr   = Web3.to_checksum_address(p['collateral_address'])
    coll_wei    = p['collateral_amount_wei']
    borrow_token= p['borrow_token']
    borrow_addr = Web3.to_checksum_address(p['borrow_address'])
    ltv         = random.uniform(p['ltv_min'], p['ltv_max'])

    # 1. Acquire collateral
    if coll_token == 'WETH':
        _swap.wrap_eth(coll_wei)
    else:
        _swap.attempt_swap(_swap.swap_eth_to_token, coll_addr, coll_wei)
    time.sleep(2)

    # 2. Approve + supply collateral
    executor._approve_if_needed(coll_addr, pool_addr, coll_wei)
    tx = pool.functions.supply(
        coll_addr, coll_wei, executor.WALLET, 0
    ).build_transaction(executor._tx_params())
    tx['gas'] = executor._gas_limit(tx)
    executor._send(tx)
    time.sleep(3)

    # 3. Explicitly enable collateral (best practice — avoids silent mode issues)
    tx2 = pool.functions.setUserUseReserveAsCollateral(
        coll_addr, True
    ).build_transaction(executor._tx_params())
    tx2['gas'] = executor._gas_limit(tx2)
    executor._send(tx2)
    time.sleep(2)

    # 4. Compute borrow amount
    borrow_wei = _compute_borrow_wei(p, ltv)
    log.info(f'AAVE borrow: coll={coll_token} {coll_wei} ltv={ltv:.2%} borrow={borrow_token} {borrow_wei}')

    # 5. Borrow
    tx3 = pool.functions.borrow(
        borrow_addr, borrow_wei, 2, 0, executor.WALLET
    ).build_transaction(executor._tx_params())
    tx3['gas'] = executor._gas_limit(tx3)
    borrow_txh = executor._send(tx3)
    time.sleep(3)

    # 6. Release borrow token → ETH
    if borrow_token == 'WETH':
        _swap.unwrap_all_weth()
    else:
        c = executor.w3.eth.contract(address=borrow_addr, abi=executor.ERC20_ABI)
        bal = c.functions.balanceOf(executor.WALLET).call()
        if bal > 0:
            _swap.attempt_swap(_swap.swap_token_to_eth, borrow_addr, bal)

    return encode_state(coll_token, coll_wei, borrow_token, borrow_wei), borrow_txh


# ── Close borrow ───────────────────────────────────────────────────────────────

def close_borrow(encoded_state: str, p: dict) -> str:
    """
    Close AAVE borrow position: repay debt → withdraw collateral → release → ETH.
    Returns tx hash of final collateral withdrawal.

    Steps:
      1. Read actual vDebt balance (includes accrued interest)
      2. Acquire repay_amount = vdebt_balance × REPAY_BUFFER
      3. Approve Pool for repay_amount
      4. Pool.repay(borrow_asset, MAX_UINT256, 2, wallet)  ← repays all vDebt
      5. Pool.withdraw(collateral, MAX_UINT256, wallet)     ← withdraws all aToken
      6. Swap collateral → ETH (or unwrap WETH)
    """
    executor._guard()
    pool = _pool()
    pool_addr = Web3.to_checksum_address(POOL_ADDR)

    coll_token, coll_wei, borrow_token, borrow_wei = decode_state(encoded_state)
    coll_addr   = Web3.to_checksum_address(p['collateral_address'])
    borrow_addr = Web3.to_checksum_address(p['borrow_address'])
    vdebt_addr  = Web3.to_checksum_address(p['borrow_vdebt'])

    # 1. Read actual debt
    vdebt_c      = executor.w3.eth.contract(address=vdebt_addr, abi=_VDEBT_ABI)
    actual_debt  = vdebt_c.functions.balanceOf(executor.WALLET).call()
    if actual_debt == 0:
        log.warning('AAVE close_borrow: vDebt balance=0, nothing to repay')
        actual_debt = borrow_wei  # fallback: use stored value

    repay_amount = int(actual_debt * REPAY_BUFFER)
    log.info(f'AAVE repay: {borrow_token} actual_debt={actual_debt} repay_amount={repay_amount}')

    # 2. Acquire repay token
    if borrow_token == 'WETH':
        _swap.wrap_eth(repay_amount)
    else:
        _swap.attempt_swap(_swap.swap_eth_to_token, borrow_addr, repay_amount)
    time.sleep(3)

    # 3. Approve + repay
    executor._approve_if_needed(borrow_addr, pool_addr, repay_amount)
    tx = pool.functions.repay(
        borrow_addr, MAX_UINT256, 2, executor.WALLET
    ).build_transaction(executor._tx_params())
    tx['gas'] = executor._gas_limit(tx)
    executor._send(tx)
    time.sleep(4)

    # 4. Withdraw all collateral
    tx2 = pool.functions.withdraw(
        coll_addr, MAX_UINT256, executor.WALLET
    ).build_transaction(executor._tx_params())
    tx2['gas'] = executor._gas_limit(tx2)
    withdraw_txh = executor._send(tx2)
    time.sleep(4)

    # 5. Release collateral → ETH
    if coll_token == 'WETH':
        _swap.unwrap_all_weth()
    else:
        c = executor.w3.eth.contract(address=coll_addr, abi=executor.ERC20_ABI)
        bal = c.functions.balanceOf(executor.WALLET).call()
        if bal > 0:
            _swap.attempt_swap(_swap.swap_token_to_eth, coll_addr, bal)

    # Sweep any leftover repay token (if REPAY_BUFFER bought excess)
    if borrow_token != 'WETH':
        bc = executor.w3.eth.contract(address=borrow_addr, abi=executor.ERC20_ABI)
        leftover = bc.functions.balanceOf(executor.WALLET).call()
        if leftover > 0:
            try:
                _swap.attempt_swap(_swap.swap_token_to_eth, borrow_addr, leftover)
            except Exception:
                pass  # leftover is small, best-effort only

    return withdraw_txh
```
